\begin{exercise}

\outcome{Find domain and range.}

\tag{domain}

For the following functions, state the domain of the given function.
\begin{enumerate}
  \begin{multicols}{3}
\item $f(x) = 2-x$
\item $f(x) = \dfrac{x - 2}{3}$
\item $f(x) = x^2 + 1$
\item $f(x) = 4-x^2$
\item $f(x) = 2$
\item $f(x) = x^3$
\item $f(x) = x(x-1)$
\item $f(x) = \sqrt{x-2}$
\item $f(x) = \sqrt{5 - x}$
\item $f(x) = 3-2\sqrt{x+2}$
\item $f(x) = \sqrt[3]{x}$
\item $f(x) = \dfrac{1}{x^{2} + 1}$
  \end{multicols}
\end{enumerate}

Answer:
\begin{enumerate}
  \begin{multicols}{3}
  \item $(\answer{-\infty},\answer{\infty})$
  \item $(\answer{-\infty},\answer{\infty})$
  \item $(\answer{-\infty},\answer{\infty})$
  \item $(\answer{-\infty},\answer{\infty})$
  \item $(\answer{-\infty},\answer{\infty})$
  \item $(\answer{-\infty},\answer{\infty})$
  \item $(\answer{-\infty},\answer{\infty})$
  \item $[\answer{2},\answer{\infty})$
  \item $(\answer{-\infty},\answer{5}]$
  \item $[\answer{-2},\answer{\infty})$
  \item $(\answer{-\infty},\answer{\infty})$
  \item $(\answer{-\infty},\answer{\infty})$
  \end{multicols}
\end{enumerate}

\end{exercise}
