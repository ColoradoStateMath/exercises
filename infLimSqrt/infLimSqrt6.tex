\begin{exercise}
\outcome{Calculate the limit as $x$ approaches $\pm \infty$ of common functions algebraically.}
\outcome{Compute limits at infinity of famous functions.}
\tag{limit}
Let 
\[
F(z) = \frac{\sqrt{z^4-4}+3 z+1}{2 z^2+3 z+4}.
\]
Compute
\begin{enumerate}
\item $\lim_{z\to \infty} F(z) \begin{prompt} = \answer{\frac{1}{2}}\end{prompt}$
\item $\lim_{z\to -\infty}F(z) \begin{prompt} = \answer{\frac{1}{2}}\end{prompt}$
\end{enumerate}
\begin{hint}
Multiply by
\[
\frac{\frac{1}{z^2}}{\frac{1}{z^2}}
\]
\end{hint}
\end{exercise}
