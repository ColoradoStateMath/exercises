\begin{exercise}
\outcome{Calculate the limit as $x$ approaches $\pm \infty$ of common functions algebraically.}
\outcome{Compute limits at infinity of famous functions.}
\tag{limit}
Let 
\[
y(\psi) = \frac{3 \psi ^3+3 \psi ^2}{4 \psi ^2+\sqrt{3 \psi ^6+3 \psi ^5+4 \psi ^4-4 \psi }-2}.
\]
Compute
\begin{enumerate}
\item $\lim_{\psi\to \infty} y(\psi) \begin{prompt} = \answer{\sqrt{3}}\end{prompt}$
\item $\lim_{\psi\to -\infty}y(\psi) \begin{prompt} = \answer{-\sqrt{3}}\end{prompt}$
\end{enumerate}
\begin{hint}
Multiply by
\[
\frac{\frac{1}{\psi ^3}}{\frac{1}{\psi ^3}}
\]
\end{hint}
\end{exercise}
