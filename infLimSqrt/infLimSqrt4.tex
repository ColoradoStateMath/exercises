\begin{exercise}
\outcome{Calculate the limit as $x$ approaches $\pm \infty$ of common functions algebraically.}
\outcome{Compute limits at infinity of famous functions.}
\tag{limit}
Let 
\[
r(k) = \frac{\sqrt{k^6+5}-2 k^2}{k-k^3}.
\]
Compute
\begin{enumerate}
\item $\lim_{k\to \infty} r(k) \begin{prompt} = \answer{-1}\end{prompt}$
\item $\lim_{k\to -\infty}r(k) \begin{prompt} = \answer{1}\end{prompt}$
\end{enumerate}
\begin{hint}
Multiply by
\[
\frac{\frac{1}{k^3}}{\frac{1}{k^3}}
\]
\end{hint}
\end{exercise}
