\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\tag{limit}
Consider 
\[
f(x) = \frac{-3}{z+1}.
\]
Compute
\begin{enumerate}
\item $\lim_{z\to -1^- } f(x) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{z\to -1^+ } f(x) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{z\to -1 } f(x) \begin{prompt} = \answer{DNE}\end{prompt}$
\end{enumerate}
\end{exercise}
