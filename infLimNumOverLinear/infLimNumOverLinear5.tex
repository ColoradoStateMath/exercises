\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\tag{limit}
Consider 
\[
f(\psi) = \frac{-5}{\psi -4}.
\]
Compute
\begin{enumerate}
\item $\lim_{\psi\to 4^- } f(\psi) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{\psi\to 4^+ } f(\psi) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{\psi\to 4 } f(\psi) \begin{prompt} = \answer{DNE}\end{prompt}$
\end{enumerate}
\end{exercise}
