\begin{exercise}

\outcome{Use the first derivative to determine whether a function is increasing or decreasing.}
\outcome{Understand what information the derivative gives concerning when a function is increasing or decreasing.}
\outcome{Find domain and range.}
\outcome{Find critical points.}
\outcome{Find all local maximums and minimums using the 1st and 2nd derivative tests.}
\outcome{Classify critical points.}




\tag{increasing}
\tag{decreasing}
\tag{derivative test}

Let $f(x)=\sin(x)\cos(x)$.
\begin{enumerate}
\item		The domain of $f$ is $\begin{prompt}\answer{(-\infty,\infty)}\end{prompt}$.
\item		On the interval $(-\pi,\pi)$, $f$ has a critical point(s) at $x = \answer{\{-3\pi/4,-\pi/4,\pi/4,3\pi/4\}}$, write DNE if no such point(s) exists.
\item		On the interval $(-\pi,\pi)$, $f$ is decreasing on $\answer{(-3\pi/4,-\pi/4)\cup(\pi/4,3\pi/4)}$, write DNE if no such interval(s) exist.
\item		On the interval $(-\pi,\pi)$, $f$ is increasing on $\answer{(-\pi,-3\pi/4)\cup(-\pi/4,\pi/4)\cup(3\pi/4,\pi)}$, write DNE if no such interval(s) exist.

\item		For $x=-3\pi/4$ and $x=\pi/4$, $f$ has 
\begin{multipleChoice}
\choice{A local minimum}
\choice[correct]{A local maximum}
\choice{Both a local maximum and a local minimum}
\choice{Neither a local maximum nor a local minimum}
\end{multipleChoice}

\item		For $x=-\pi/4$ and $x=3\pi/4$, $f$ has 
\begin{multipleChoice}
\choice[correct]{A local minimum}
\choice{A local maximum}
\choice{Both a local maximum and a local minimum}
\choice{Neither a local maximum nor a local minimum}
\end{multipleChoice}
\end{enumerate}

\end{exercise}
