\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}

\[\lim_{x \to 0} \frac{\sin(2x)}{3x} = \answer{2/3}\]

\begin{hint}
	\[\frac{\sin(2x)}{3x} = \frac{\sin(2x)}{2x} \frac{2}{3}\]
\end{hint}

\end{exercise}
