\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
	
	\begin{warning}
		This problem is pretty hard.
	\end{warning}
	
	\[\lim_{h \to 0} \frac{\sin(1+h) - 2\sin(1)+\sin(1-h)}{h^2} = \answer{-\sin(1)}\]
	
	\begin{hint}
		Use the angle sum formula for sine on both $\sin(1+h)$ and $\sin(1-h)$
	\end{hint}


\end{exercise}
