\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}

\begin{question}
	\[\lim_{h \to 0} \frac{\sqrt{2+h} - \sqrt{2}}{h} = \answer{1/(2\sqrt{2})}\]
	
	\begin{hint}
		Multiply numerator and denominator by $\sqrt{2+h}+\sqrt{2}$.
	\end{hint}

\end{exercise}
