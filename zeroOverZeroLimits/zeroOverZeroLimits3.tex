\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}

\[\lim_{x \to 0} \frac{\sin(5x)}{\sin(2x)} = \answer{5/2}\]

\begin{hint}
	\[ \frac{\sin(5x)}{\sin(2x)} = \frac{\sin(5x)}{5x} \frac{2x}{\sin(2x)} \frac{5}{2}\]
\end{hint}

\end{exercise}
