\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $g(\psi) = \psi ^2$. Compute

\[
\lim_{ \psi\to 4 } 
\frac{g(\psi)-g(4)}{\psi -4} \begin{prompt}=\answer{8}\end{prompt}
\]
