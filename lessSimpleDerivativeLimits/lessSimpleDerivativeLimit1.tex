\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $R(\theta) = \frac{1}{\theta +8}$. Compute

\[
\lim_{ \theta\to 4 } 
\frac{R(\theta)-R(4)}{\theta -4} \begin{prompt}=\answer{-\frac{1}{144}}\end{prompt}
\]
