\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $A(\psi) = \frac{1}{\psi +4}$. Compute

\[
\lim_{ \psi\to 2 } 
\frac{A(\psi)-A(2)}{\psi -2} \begin{prompt}=\answer{-\frac{1}{36}}\end{prompt}
\]
