\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $s(\psi) = \frac{1}{\psi +2}$. Compute

\[
\lim_{ \psi\to -1 } 
\frac{s(\psi)-s(-1)}{\psi +1} \begin{prompt}=\answer{-1}\end{prompt}
\]
\end{exercise}
