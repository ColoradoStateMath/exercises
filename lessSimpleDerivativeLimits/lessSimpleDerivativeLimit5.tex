\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $Y(z) = \frac{1}{z+5}$. Compute

\[
\lim_{ z\to -2 } 
\frac{Y(z)-Y(-2)}{z+2} \begin{prompt}=\answer{-\frac{1}{9}}\end{prompt}
\]
\end{exercise}
