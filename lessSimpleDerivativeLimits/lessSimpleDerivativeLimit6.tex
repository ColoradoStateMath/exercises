\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $p(x) = \sqrt{x+4}$. Compute

\[
\lim_{ x\to 2 } 
\frac{p(x)-p(2)}{x-2} \begin{prompt}=\answer{\frac{1}{2 \sqrt{6}}}\end{prompt}
\]
