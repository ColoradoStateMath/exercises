\begin{exercise}

\outcome{Compute limits of sequences.}
\outcome{Understand growth rates of basic sequences.}

\tag{sequence}

Determine whether the sequence $a_n=\frac{4n^2-n+5}{3n^2+1}$ $(n=1,2,3,\ldots)$ converges or diverges. If you conclude that it diverges, write DNE.

\[
\begin{prompt}
\lim_{n\to\infty}a_n=\answer{4/3}
\end{prompt}
\]

\end{exercise}
