\outcome{Use partial fractions to integrate functions.}
\outcome{Recognize integrals that are good candidates for the method of partial fractions.}

\tag{partial fractions}
\tag{integration}
\tag{antiderivative}
\tag{indefinite integral}



Find the indefinite integral of $\frac{x^2-20x-69}{(x-7)(x^2+2x+17)}$ with respect to $x$.
\[
\begin{prompt}
\int \frac{x^2-20x-69}{(x-7)(x^2+2x+17)} \d x=\answer{\frac{1}{2} \left(3 \ln(|x^2+2 x+17|)-4\ln(|x-7|)+\arctan\left(\frac{x+1}{4}\right)\right) +C}\end{prompt}
\]
