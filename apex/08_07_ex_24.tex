\begin{exercise}

\outcome{Determine the maximum error between a function and a given Taylor polynomial.}
\outcome{Compute Taylor polynomials.}

\tag{power series}
\tag{taylor polynomial}



Find the smallest integer $n$ such that the Taylor polynomial of degree $n$ of $f(x)=\sqrt{x}$, centered at $x=4$, approximates $\sqrt{3}$ within $0.0001$ of the actual value. 

\[
\begin{prompt}
n=\answer{4}
\end{prompt}
\]
