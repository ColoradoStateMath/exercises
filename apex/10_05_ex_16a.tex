\begin{exercise}

\outcome{Answer questions about lines and curves in space.}

\tag{curves}


Let $\vec{\ell}_{1}(t)=\vector{2,1,1}+t\vector{5,1,3}$ and $\vec{\ell}_{2}(t)=\vector{14,5,9}+t\vector{1,1,1}$. 

Then $\vec{\ell}_{1}$ and $\vec{\ell}_{2}$ are: 

\begin{multipleChoice}
\choice[correct]{parallel}
\choice{intersecting}
\choice{non-intersecting}
\choice{the same}
\end{multipleChoice}

\end{exercise}
