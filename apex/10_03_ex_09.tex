\begin{exercise}

\outcome{Compute dot products.}

\tag{dot products}
\tag{vectors}
\tag{true or false}

Is the following statement true or false?
\begin{quote}
The dot product of a vector in $\mathbb{R}^2$ and a vector in $\mathbb{R}^3$ is not defined.
\end{quote}

\begin{multipleChoice}
 \choice[correct]{True}
 \choice{False}
\end{multipleChoice}

\end{exercise}
