\outcome{Compute Taylor polynomials.}

\tag{power series}
\tag{taylor polynomial}


Find the Taylor polynomial of degree $7$ of $f(x)=\arctan x$ centered at $x=0$.

\[
\begin{prompt}
p_{7}(x)=\answer{-\frac{x^7}{7}+\frac{x^5}{5}-\frac{x^3}{3}+x}
\end{prompt}
\]
