\outcome{Use the divergence test to determine if a series diverges or converges.}

\tag{series}
\tag{true or false}

True or false?

\begin{quote}
If the sequence $(a_n)$ converges, then the sum $\sum_{n=1}^{\infty} a_n$ converges.
\end{quote}

\begin{multipleChoice}
\choice{True}
\choice[correct]{False}
\end{multipleChoice}
