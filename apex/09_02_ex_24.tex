\begin{exercise}

\outcome{Eliminate the parameters of a parametric equation.}

\tag{parametric}


Eliminate the parameters of the parametrization given by $x(t)=e^t$ and $y(t)=e^{3t}-3$. You can leave $x$ and $y$ implicitly defined, if need be.

\[
\begin{prompt}
y=\answer{x^3-3}
\end{prompt}
\]

\end{exercise}
