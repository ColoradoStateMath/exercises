\begin{exercise}


\outcome{Use the divergence test to determine that a series diverges.}

\tag{series}
\tag{series test}


The series $\sum_{n=1}^{\infty} n$ does not converge. Which of the following best explains why?

\begin{multipleChoice}
\choice{The sum is a geometric series.}
\choice{The sum is a telescoping series.}
\choice{The integral test.}
\choice[correct]{The divergence test.}
\choice{The ratio test}
\choice{The root test}
\choice{The comparison test.}
\end{multipleChoice}
\end{exercise}
