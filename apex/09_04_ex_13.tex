\begin{exercise}

\outcome{Convert between polar and Cartesian coordinates.}
\outcome{Use the Cartesian to polar method to plot polar graphs.}


\tag{polar coordinates}


Given the graph of a function in rectangular (Cartesian) coordinates below, find the equation representing the graph in polar coordinates subject to the correct restriction(s) on $\theta$ and $r$.

 \begin{center}
 \begin{tikzpicture}
 \begin{axis}[ymin=-2.1, ymax=2.1,
   axis lines=center, 
   axis equal,
   xlabel=$x$, 
   ylabel=$y$,
   every axis y label/.style={at=(current axis.above origin),anchor=south}, 
   every axis x label/.style={at=(current axis.right of origin),anchor=west}, 
    xticklabels={,,,,,$2$},
    yticklabels={,,,,,$2$},
    grid = major]
  \addplot[penColor, very thick, samples=400,domain=0:2*pi,
    data cs=polarrad] (x,{1-cos(deg(x))});
 \end{axis} 
\end{tikzpicture}     
\end{center} 

\[
r(\theta)\begin{prompt}=\answer{1-\cos(\theta)}\end{prompt}
\]
subject to the restriction(s) that
\[
0\le\theta\le\begin{prompt}\answer{2\pi}\end{prompt}
\]

      
\end{exercise}
