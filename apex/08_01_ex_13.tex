\begin{exercise}

\outcome{Compute limits of sequences.}
\outcome{Understand growth rates of basic sequences.}


\tag{sequence}

Determine whether the sequence $a_n=(-1)^n\frac{n}{n+1}$ $(n=1,2,3,\ldots)$ converges or diverges. If you conclude that it diverges, write DNE.

\[
\begin{prompt}
\lim_{n\to\infty}a_n=\answer{DNE}
\end{prompt}
\]
