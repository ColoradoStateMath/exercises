\outcome{Compute Taylor polynomials.}

\tag{power series}

\[
\begin{prompt}
p_{4}(x)=\answer{\frac{2}{3}x^4+\frac{4}{3}x^3+2x^2+2x+1}
\end{prompt}
\]
