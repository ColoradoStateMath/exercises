\begin{exercise}

\outcome{Compute Taylor polynomials.}


\tag{power series}
\tag{taylor polynomial}


Find the Taylor polynomial of degree $2$ of $f(x)=x^2\cos x$ centered at $x=\pi$.

\[
\begin{prompt}
p_{2}(x)=\answer{-\pi^2-2 \pi  (x-\pi)+\frac{1}{2} \left(\pi ^2-2\right)(x-\pi)^2}
\end{prompt}
\]

\end{exercise}
