\begin{exercise}

\outcome{Eliminate the parameters of a parametric equation.}


\tag{parametric}


Eliminate the parameters of the parametrization given by $x(t)=\cos(2t)$ and $y(t)=\sin(t)$. You can leave $x$ and $y$ implicitly defined, if need be.

\[
\begin{prompt}
x=\answer{1-2y^2}
\end{prompt}
\]

\end{exercise}
