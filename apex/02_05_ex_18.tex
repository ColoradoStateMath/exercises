\begin{exercise}

\tag{derivative}

\outcome{Apply the generalized power rule.}
Find the derivative of $g(r)=4^r$ with respect to $r$.

\[
g'(r)
\begin{prompt}
= \answer{\ln(4)\cdot 4^r}
\end{prompt}
\]
