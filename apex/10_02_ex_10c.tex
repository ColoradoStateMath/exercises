\begin{exercise}

\outcome{Work with vectors in two or three dimensions.}
\outcome{Add and subtract vectors.}
\outcome{Use vectors in applied settings.}
\outcome{Multiply vectors by scalars.}

\tag{vectors}


Let $\mathbf{u}=\left\langle1,-2\right\rangle$ and $\mathbf{v}=\left\langle1,1\right\rangle$ be two vectors. Find the vector $2\mathbf{u}-3\mathbf{v}$.

\[
\begin{prompt}
2\mathbf{u}-3\mathbf{v}=\left\langle\answer{-1},\answer{-7}\right\rangle
\end{prompt}
\]
