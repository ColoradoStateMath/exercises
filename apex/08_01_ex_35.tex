\begin{exercise}

\outcome{Compute limits of sequences.}
\outcome{Understand growth rates of basic sequences.}

\tag{sequence}

Determine whether the sequence $a_n= \frac{2^n-20}{7\cdot2^n} $ $(n=1,2,3,\ldots)$ converges or diverges. If you conclude that it diverges, write DNE.

\[
\begin{prompt}
\lim_{n\to\infty}a_n=\answer{\frac{1}{7}}
\end{prompt}
\]


\end{exercise}
