\begin{exercise}

\outcome{Compute derivatives of polar curves.}
\outcome{Find horizontal and vertical tangent lines of a polar curve.}
\outcome{Determine where the derivative of a polar curve is undefined.}


\tag{polar coordinates}
\tag{derivative}


Find all values of $\theta$ in $[0,2\pi]$ for which the tangent line to the polar curve $r(\theta)=1+\cos\theta$ is horizontal. List your values of $\theta$ from least to greatest. 
\[
\begin{prompt}\text{$\theta$ is one of } \answer{\frac{\pi}{3}}, \answer{\pi}, \answer{\frac{5\pi}{3}}\ldotp\end{prompt}
 \]
 
Conversely, find all values of $\theta$ in $[0,2\pi]$ for which the tangent line to the polar curve $r(\theta)=1+\cos\theta is vertical. List your values of $\theta$ from least to greatest. 
 \[
\begin{prompt}\text{$\theta$ is one of } \answer{0}, \answer{\frac{2\pi}{3}}, \answer{\frac{4\pi}{3}}, \answer{2\pi}\ldotp\end{prompt}
 \]

 
\end{exercise}
