\begin{exercise}

\outcome{Given a sequence find an explicit formula.}

\tag{sequence}

Let  $10, 20, 40, 80, 160, \ldots$ be a sequence. If $a_n$ denotes the $n$th member of this sequence, what is the rule that describes $a_n$?
\[
a_n
\begin{prompt}
=\answer{10\cdot 2^{n-1}}
\end{prompt}
\]

\end{exercise}
