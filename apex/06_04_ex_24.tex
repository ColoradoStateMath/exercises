\outcome{Use partial fractions to integrate functions.}
\outcome{Recognize integrals that are good candidates for the method of partial fractions.}

\tag{partial fractions}
\tag{integration}
\tag{antiderivative}
\tag{indefinite integral}



Find the indefinite integral of $\frac{9x^2-60x+33}{(x-9)(x^2-2x+11)}$ with respect to $x$.
\[
\begin{prompt}
\int \frac{9x^2-60x+33}{(x-9)(x^2-2x+11)} \d x=\answer{3 \left(\ln(|x^2-2 x+11|)+\ln(|x-9|)+3\sqrt{\frac{2}{5}}\arctan\left(\frac{x-1}{\sqrt{10}}\right)\right) +C}
\end{prompt}
\]
