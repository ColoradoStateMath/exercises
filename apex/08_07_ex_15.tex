\begin{exercise}

\outcome{Compute Taylor polynomials.}

\tag{power series}
\tag{taylor polynomial}


Find the Taylor polynomial of degree $3$ of $f(x)=\frac{1}{x^2+1}$ centered at $x=-1$.

\[
\begin{prompt}
p_{3}(x)=\answer{\frac{1}{2}+\frac{1+x}{2}+\frac{1}{4} (1+x)^2}
\end{prompt}
\]
