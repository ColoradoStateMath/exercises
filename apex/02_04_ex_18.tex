\begin{exercise}

\tag{derivative}
\tag{product rule}

\outcome{Identify quotients of functions.}
\outcome{Use the quotient rule to calculate derivatives.}




Find the derivative of $g(x)=\frac{x+7}{x-5}$.
\[
g'(x)
\begin{prompt}
= \answer{-\frac{12}{(x-5)^2}}
\end{prompt}
\]
