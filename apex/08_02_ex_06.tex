\begin{exercise}


\outcome{Use the integral test to determine if a series diverges or converges.}


\tag{series}
\tag{series test}


The series $\sum_{n=1}^{\infty} \frac{1}{n^2}$ converges. Which of the following best explains why?

\begin{multipleChoice}
\choice{The sum is a geometric series.}
\choice{The sum is a telescoping series.}
\choice[correct]{The integral test.}
\choice{The divergence test.}
\choice{The ratio test}
\choice{The root test}
\choice{The comparison test.}
\end{multipleChoice}
\end{exercise}
