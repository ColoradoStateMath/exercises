\begin{exercise}
\outcome{Recognize a composition of functions.}
\outcome{Take derivatives of compositions of functions using the chain rule.}
\outcome{Use the chain rule to calculate derivatives from a table of values.}

\tag{derivative}

Let $f(x) = g(\sqrt(x))$, where $g$ is a function which has the properties shown in the table below.  Evaluate $f'(5)$.

\[
\begin{array}{c|c|c}
 x & g(x) & g'(x)\\ \hline
1 & -3 & 4\\
2 & 2 & 3\\
3 & -1 & 6\\
4 & -7 & 9\\
5 & 8 & 1\\
6 & -4 & 6\\
7 & -5 & -5\\
8 & 6 & -2\\
9 & 8 & 0\\
10 & -6 & 1\\
\end{array}
\]

\begin{prompt}
	$$f'(5) = \answer{-54}$$
\end{prompt}


\end{exercise}
