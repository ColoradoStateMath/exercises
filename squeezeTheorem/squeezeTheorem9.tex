\begin{exercise}

\outcome{Understand the Squeeze Theorem and how it can be used to find limit values.}
\outcome{Calculate limits using the Squeeze Theorem.}
\tag{limit}
\tag{squeeze theorem}

Let $f$ be a function which satisfies the following inequality:

\[
\frac{x^2- 9}{x - 3} \leq f(x) \leq \frac{x^2+x-12}{x-3}
\]

for all $x >3$.

\begin{multipleChoice}
	\choice{We can use this information to determine $\lim_{x \to 3^+}f(x)$}
	\choice{We can conclude that $\lim_{x \to 3^+} f(x)$ exists, but cannot determine its value}
	\choice[correct]{We do not know whether $\lim_{x \to 3^+} f(x)$ exists}
\end{multipleChoice}
\end{exercise}
