\begin{exercise}

\outcome{Understand the Squeeze Theorem and how it can be used to find limit values.}
\outcome{Calculate limits using the Squeeze Theorem.}
\tag{limit}
\tag{squeeze theorem}

Note that 
	
	\[4 \leq (3^{\frac{1}{x}}+ 4^{\frac{1}{x}})^x \leq  (2 (4^{\frac{1}{x}})^x\].
	
	whenever $x>0$
	
	Use this, together with the squeeze theorem, to find $\lim_{x \to 0} (3^{\frac{1}{x}}+ 4^{\frac{1}{x}})^x$
	
	\[\lim_{x \to 0} (3^{\frac{1}{x}}+ 4^{\frac{1}{x}})^x = \answer{4}\]

\end{exercise}
