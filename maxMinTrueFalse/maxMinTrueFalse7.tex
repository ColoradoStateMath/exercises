\begin{exercise}
\outcome{Define local maximum and local minimum.}
\tag{derivative}

	If $f$ and $g$ both have local minima at $x=a$, then their sum $f+g$ has a local minima at $x=a$ as well.
	\begin{multipleChoice}	
		\choice[correct]{True}
		\choice{False}
	\end{multipleChoice}
