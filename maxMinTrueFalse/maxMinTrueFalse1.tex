\begin{exercise}
\outcome{ Define a critical point.}
\outcome{ Find critical points.}
\outcome{ Define absolute maximum and absolute minimum.}
\outcome{ Find the absolute max or min of a continuous function on a closed interval.}
\outcome{ Define local maximum and local minimum.}
\outcome{ Compare and contrast local and absolute maxima and minima.}
\outcome{ Identify situations in which an absolute maximum or minimum is guaranteed.}
\outcome{ Classify critical points.}
\outcome{ State the First Derivative Test.}
\outcome{ Apply the First Derivative Test.}
\tag{derivative}


The following questions test your \textbf{precise} understanding of the terms involved.  Some of these are tricky.

\begin{question}
	The same point can be both a local max and a local min for the same function.
	\begin{multipleChoice}	
		\choice[correct]{True}
		\choice{False}
	\end{multipleChoice}
\end{question}

\begin{question}
	A function on a \textbf{closed} interval must have a local extremum.
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}
\end{question}

\begin{question}
	If $f'(x)<0$ when $x<a$ and $f'(x) \geq 0$ when $x>a$, then $a$ is a local max.
	\begin{multipleChoice}	
		\choice[correct]{True}
		\choice{False}
	\end{multipleChoice}
\end{question}

\begin{question}
	If $f$ has a local max at $x=a$, then its square $f^2$ must also have a local max at $x=a$
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}
\end{question}

\begin{question}
	If $f'(a)$ does not exist, then $f$ cannot have a local extremum at $x=a$
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}
\end{question}

\begin{question}
	If $f$ and $g$ both have local minima at $x=a$, then their sum $f+g$ has a local minima at $x=a$ as well.
	\begin{multipleChoice}	
		\choice[correct]{True}
		\choice{False}
	\end{multipleChoice}
\end{question}

\begin{question}
	If $f''(x)<0$ on $(2,5)$, and $f'(3)=0$, then $x=3$ is a local max
	\begin{multipleChoice}	
		\choice[true]{True}
		\choice{False}
	\end{multipleChoice}
\end{question}

\begin{question}
	If $f$ is differentiable and decreasing on $(a,b)$, then $f'(x)<0$ on $(a,b)$
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}
\end{question}

\begin{question}
	If $f$ and $g$ both have local minima at $x=a$, then their product $fg$ has a local minima at $x=a$ as well.
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}
\end{question}

\begin{question}
	If $f'(a) = 0$, then $x=a$ is a local extremum of $f$.
	\begin{multipleChoice}	
		\choice{True}
		\choice[correct]{False}
	\end{multipleChoice}
\end{question}



\end{exercise}
