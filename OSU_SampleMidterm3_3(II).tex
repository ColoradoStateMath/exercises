\begin{exercise}

\outcome{Find the linear approximation to a function at a point and use it to approximate the function value.}
\outcome{Define linear approximation as an application of the tangent to a curve.}

Find the equation of the line that represents the linear approximation to the function $f(x)=e^{2x}$ at $x=0$.

\begin{prompt}
\begin{multipleChoice}
\choice[correct]{$y=1+2x$}
\choice{$y=1-2x$}
\choice{$y=\ln(2x+1)$}
\choice{$y=x$}
\choice{Such a line does not exist}
\choice{None of the previous answers}
\end{multipleChoice}
\end{prompt}
\end{exercise}
