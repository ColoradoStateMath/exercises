\begin{exercise}
\outcome{Solve basic related rates word problems.}
\outcome{Identify word problems as related rates problems.}
\outcome{Understand the process of solving related rates problems.}
\tag{related rates}
A light shines from the top of a pole 20 m high.  An object is dropped from
the same height from a point 10 m away, so that its height at time $t$
seconds is $h(t)=20-9.8t^2/2$.  How fast is the object's shadow
moving on the ground one second later?
\begin{prompt}
  \[
  \answer{4000/49}\,\text{m/s}
  \]
\end{prompt}
\end{exercise}
