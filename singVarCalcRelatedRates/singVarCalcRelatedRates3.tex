\begin{exercise}
\outcome{Solve basic related rates word problems.}
\outcome{Identify word problems as related rates problems.}
\outcome{Understand the process of solving related rates problems.}
\tag{related rates}

 A ladder 13 meters long rests on horizontal ground and leans against
 a vertical wall.  The foot of the ladder is pulled away from the wall
 at the rate of 0.6 m/sec.  How fast is the top sliding down the wall
 when the foot of the ladder is 5 m from the wall?
\begin{prompt}
  \[
  \answer{1/4}\,\text{m/s}
  \]
\end{prompt}
\end{exercise}
