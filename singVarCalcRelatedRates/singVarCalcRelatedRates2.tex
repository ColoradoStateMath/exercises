\begin{exercise}
\outcome{Solve basic related rates word problems.}
\outcome{Identify word problems as related rates problems.}
\outcome{Understand the process of solving related rates problems.}
\tag{related rates}
A cylindrical tank standing upright (with one circular base on the
ground) has radius 1 meter.  How fast does the water level in the
tank drop when the water is being drained at 3 liters per second?
\begin{prompt}
  \[
  \answer{3/(1000\pi)}\,\text{m/s}
  \]
\end{prompt}
\end{exercise}
