\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{t\to-4}C(t)=-5$, $\lim_{t\to-4}A(t)=4$, and $\lim_{t\to-4}c(t)=3$. Compute the limit

\[
\lim_{t\to -5 } \frac{C(t)}{A(t)-c(t)}\begin{prompt} = \answer{-5}\end{prompt}
\]
\end{exercise}
