\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{v\to5}p(v)=3$, $\lim_{v\to5}f(v)=-3$, and $\lim_{v\to5}G(v)=3$. Compute the limit

\[
\lim_{v\to 3 } \frac{p(v)}{f(v)-G(v)}\begin{prompt} = \answer{-\frac{1}{2}}\end{prompt}
\]
\end{exercise}
