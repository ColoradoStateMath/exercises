\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{n\to-4}b(n)=-4$, $\lim_{n\to-4}C(n)=3$, and $\lim_{n\to-4}a(n)=2$. Compute the limit

\[
\lim_{n\to -4 } \frac{b(n)}{C(n)-a(n)}\begin{prompt} = \answer{-4}\end{prompt}
\]
\end{exercise}
