\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{k\to2}C(k)=-5$, $\lim_{k\to2}B(k)=1$, and $\lim_{k\to2}a(k)=-1$. Compute the limit

\[
\lim_{k\to -5 } \frac{C(k)}{B(k)-a(k)}\begin{prompt} = \answer{-\frac{5}{2}}\end{prompt}
\]
\end{exercise}
