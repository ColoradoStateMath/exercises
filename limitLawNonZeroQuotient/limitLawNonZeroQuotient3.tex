\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{u\to2}c(u)=-5$, $\lim_{u\to2}B(u)=-1$, and $\lim_{u\to2}Y(u)=-4$. Compute the limit

\[
\lim_{u\to 2 } \frac{c(u)}{B(u)-Y(u)}\begin{prompt} = \answer{-\frac{5}{3}}\end{prompt}
\]
\end{exercise}
