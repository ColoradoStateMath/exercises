\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{z\to5}G(z)=3$, $\lim_{z\to5}C(z)=-4$, and $\lim_{z\to5}c(z)=1$. Compute the limit

\[
\lim_{z\to 5 } \frac{G(z)}{C(z)-c(z)}\begin{prompt} = \answer{-\frac{3}{5}}\end{prompt}
\]
\end{exercise}
