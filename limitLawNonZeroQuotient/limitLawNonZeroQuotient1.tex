\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{v\to5}g(v)=5$, $\lim_{v\to5}g(v)=3$, and $\lim_{v\to5}Y(v)=4$. Compute the limit

\[
\lim_{v\to 5 } \frac{g(v)}{g(v)-Y(v)}\begin{prompt} = \answer{-5}\end{prompt}
\]
\end{exercise}
