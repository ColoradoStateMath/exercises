\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{z\to-3}y(z)=-4$, $\lim_{z\to-3}a(z)=2$, and $\lim_{z\to-3}P(z)=3$. Compute the limit

\[
\lim_{z\to -4 } \frac{y(z)}{a(z)-P(z)}\begin{prompt} = \answer{4}\end{prompt}
\]
\end{exercise}
