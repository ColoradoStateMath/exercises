\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{k\to5}s(k)=2$, $\lim_{k\to5}y(k)=1$, and $\lim_{k\to5}Y(k)=-3$. Compute the limit

\[
\lim_{k\to 5 } \frac{s(k)}{y(k)-Y(k)}\begin{prompt} = \answer{\frac{1}{2}}\end{prompt}
\]
\end{exercise}
