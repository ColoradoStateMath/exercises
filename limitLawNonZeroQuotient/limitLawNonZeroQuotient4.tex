\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{w\to-4}y(w)=5$, $\lim_{w\to-4}g(w)=-1$, and $\lim_{w\to-4}c(w)=-2$. Compute the limit

\[
\lim_{w\to -4 } \frac{y(w)}{g(w)-c(w)}\begin{prompt} = \answer{5}\end{prompt}
\]
\end{exercise}
