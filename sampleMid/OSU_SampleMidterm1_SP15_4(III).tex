\begin{exercise}

\outcome{Calculate limits using the limit laws.}

\tag{limits}
\tag{squeeze theorem}

Given that $\cos(x)\le f(x)\le e^x$ for all $x>0$, evaluate 
\[
\lim_{x\to0^+}f(x)
\]
\begin{prompt}
\begin{multipleChoice}
\choice{$\lim_{x\to0^+}f(x)=0$}
\choice[correct]{$\lim_{x\to0^+}f(x)=1$}
\choice{$\lim_{x\to0^+}f(x)=e$}
\choice{$\lim_{x\to0^+}f(x)$ \text{does not exist}}
\choice{We don't have enough information to answer the question}
\end{multipleChoice}
\end{prompt}
\end{exercise}
