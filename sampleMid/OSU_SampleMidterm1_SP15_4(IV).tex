\begin{exercise}

\outcome{Define and work with inverse functions.}
\outcome{Find inverse functions (algebraically and graphically).}

\tag{inverse}

Given the function $f(x)=\frac{1}{x-2}$, find its inverse, $f^{-1}(x)$.
\begin{prompt}
\begin{multipleChoice}
\choice{$f^{-1}(x)=x-2$}
\choice{$f^{-1}(x)=x+2$}
\choice{$f^{-1}(x)=\frac{1}{x}+2$}
\choice{$f^{-1}(x)=\frac{1}{x}-2$}
\choice{$f^{-1}(x)$ does not exist}
\end{multipleChoice}
\end{prompt}
\end{exercise}
