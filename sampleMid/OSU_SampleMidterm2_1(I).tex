\begin{exercise}

\tag{derivative}

\outcome{Recognize a composition of functions.}
\outcome{Compute derivatives of trig functions.}
\outcome{ Take derivatives that require the use of multiple derivative rules.}
\outcome{Take derivatives of compositions of functions using the chain rule.}
\outcome{Take derivatives of functions raised to functions.}
\outcome{Identify situations where logs can be used to help find derivatives.}



Evaluate the derivatives of the following functions.
\begin{enumerate}
\item		$\frac{d}{dx}(\sqrt{\tan(x^5+4)})\begin{prompt} = \answer{\frac{5 x^4 \sec ^{2}(x^5+4)}{2 \sqrt{\tan(x^5+4)}}}\end{prompt}$
\item		$\frac{d}{dx}(x^{(\cos(x))})\begin{prompt} = \answer{x^{\cos (x)} \left(\frac{\cos (x)}{x}-\log (x) \sin (x)\right)}\end{prompt}$
\end{enumerate}
\end{exercise}
