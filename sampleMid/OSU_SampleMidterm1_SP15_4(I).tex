\outcome{State the Intermediate Value Theorem including hypotheses.}

\tag{intermediate value theorem}

Complete the statement of the Intermediate Value Theorem:
\begin{quote}
Suppose $f$ is continuous on the interval $[a,b]$ and $L$ is a number between $f(a)$ and $f(b)$. Then there is at least one number $c$ in $(a,b)$ such that
\end{quote}

\begin{prompt}
\begin{multipleChoice}
\choice{$c=L$.}
\choice{$f(c)=0$.}
\choice[correct]{$f(c)=L$.}
\choice{$f'(c)=L$.}
\choice{$c=0$}
\choice{$f(a)<c<f(b)$}
\choice{None of the previous answers}
\end{multipleChoice}
\end{prompt}
