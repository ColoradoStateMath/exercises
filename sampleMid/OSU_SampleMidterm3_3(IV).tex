\begin{exercise}


\outcome{Add up a large number of terms quickly using sigma notation.}

Calculate the sum $\sum_{k=1}^{3}(1+k^2)$.

\begin{prompt}
\begin{multipleChoice}
\choice{$7$}
\choice{$15$}
\choice[correct]{$17$}
\choice{$\frac{k(k+1)(2k+1)}{6}$}
\choice{$16$}
\choice{None of the previous answers}
\end{multipleChoice}
\end{prompt}

\end{exercise}
