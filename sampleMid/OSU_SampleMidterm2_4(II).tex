\begin{exercise}

\tag{derivative}

\outcome{Understand what information the derivative gives concerning when a function is increasing or decreasing.}
\outcome{Find the intervals where a function is increasing or decreasing.}

The \textbf{derivative} of the function $f$ is given and a table of values for $g(x)$ and $g'(x)$ is shown below.

Suppose that $g$ is a one-to-one function and $g^{-1}(x)$ is its inverse.
\[
f'(x)=\frac{x}{x^2+1},\qquad
\begin{array}{c|c|c}
x & g(x) & g'(x)\\ \hline
1 & 2 & 4\\ \hline
2 & 3 & 5\\ \hline
3 & 4 & 1
\end{array}
\]

On what inteval(s) is $f$ increasing?
\begin{prompt}
\begin{multipleChoice}
\choice{$(-2,+\infty)$}
\choice{$(-1,+\infty)$}
\choice{$(-1,1)$}
\choice{$(-1,0)$}
\choice{$(-\infty,0)$}
\choice[correct]{$(0,+\infty)$}
\choice{None of the previous answers}
\end{multipleChoice}
\end{prompt}

\end{exercise}
