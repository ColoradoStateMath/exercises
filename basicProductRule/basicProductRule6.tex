\begin{exercise}
\outcome{Use the product rule to calculate derivatives.}
\tag{derivative}
Let $r(w) = \sqrt{w} \left(5 e^w-4 w^4\right)$. Compute
\[
\dd{w}r(w)
\begin{prompt}=\answer{\frac{5 e^w-4 w^4}{2 \sqrt{w}}+\sqrt{w} \left(5 e^w-16 w^3\right)}\end{prompt}
\]
\end{exercise}
