\begin{exercise}
\outcome{Use the product rule to calculate derivatives.}
\tag{derivative}
Let $F(u) = 10 \sin (u) (\sin (u)-4 u)$. Compute
\[
F'(u)
\begin{prompt}=\answer{10 \sin (u) (\cos (u)-4)+10 (\sin (u)-4 u) \cos (u)}\end{prompt}
\]
\end{exercise}
