\begin{exercise}

\outcome{Find the derivative function using the limit definition..}
\outcome{Relate the derivative function to the derivative at a point.}


Let $f$ be a function enjoying the following three properties:

\begin{enumerate}
 \item $f(4) = 7$
\item $f(a+b) = f(a)f(b)$ for all real numbers $a$ and $b$
\item $\lim_{x \to 0} \frac{f(x) - 1}{x} = 2$
\end{enumerate}

Compute $f'(4)$.

\begin{prompt}
	\[
	f'(4) = \answer{14}
	\]
\end{prompt}

\end{exercise}
