\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $f(x) = \sqrt{x+3}$. Compute

\[
\lim_{ x\to -1 } 
\frac{f(x)-f(-1)}{x+1} \begin{prompt}=\answer{\frac{1}{2 \sqrt{2}}}\end{prompt}
\]
\end{exercise}
