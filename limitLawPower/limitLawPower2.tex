\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{z\to5}Y(z)=-1$ and $\lim_{z\to5}f(z)=1$. Compute the limit

\[
\lim_{z\to 5 } \sqrt[10]{Y(z)f(z)+1}\begin{prompt} = \answer{1}\end{prompt}
\]
\end{exercise}
