\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{\psi\to-5}b(\psi)=-2$ and $\lim_{\psi\to-5}A(\psi)=-1$. Compute the limit

\[
\lim_{\psi\to -5 } \sqrt[2]{b(\psi)A(\psi)-1}\begin{prompt} = \answer{\sqrt{7}}\end{prompt}
\]
\end{exercise}
