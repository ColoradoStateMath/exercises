\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{k\to5}F(k)=3$ and $\lim_{k\to5}G(k)=1$. Compute the limit

\[
\lim_{k\to 5 } \sqrt[7]{F(k)G(k)+1}\begin{prompt} = \answer{\sqrt[7]{7}}\end{prompt}
\]
\end{exercise}
