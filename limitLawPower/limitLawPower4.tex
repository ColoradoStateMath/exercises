\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{z\to-1}Y(z)=4$ and $\lim_{z\to-1}R(z)=1$. Compute the limit

\[
\lim_{z\to -1 } \sqrt[2]{Y(z)R(z)+1}\begin{prompt} = \answer{1}\end{prompt}
\]
\end{exercise}
