\begin{exercise}
\outcome{Calculate the limit as $x$ approaches $\pm \infty$ of common functions algebraically.}
\outcome{Compute limits at infinity of famous functions.}
\tag{limit}
Let 
\[
y(z) = \frac{5 z^3+z^2+z+5}{2 z^3-5 z^2-3 z+3}.
\]
Compute
\begin{enumerate}
\item $\lim_{z\to \infty} y(z) \begin{prompt} = \answer{\frac{5}{2}}\end{prompt}$
\item $\lim_{z\to -\infty}y(z) \begin{prompt} = \answer{\frac{5}{2}}\end{prompt}$
\end{enumerate}
\begin{hint}
Multiply by
\[
\frac{\frac{1}{z^3}}{\frac{1}{z^3}}
\]
\end{hint}
\end{exercise}
