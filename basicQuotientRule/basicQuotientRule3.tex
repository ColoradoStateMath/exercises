\begin{exercise}
\outcome{Use the quotient rule to calculate derivatives.}
\tag{derivative}
Let $C(n) = \frac{ -\frac{4}{n^2}-3 \sin (n)}{-\sqrt{n}}$. Compute:
\[
C'(n)
\begin{prompt}=\answer{\frac{-\frac{4}{n^2}-3 \sin (n)}{2 n^{3/2}}-\frac{\frac{8}{n^3}-3 \cos (n)}{\sqrt{n}}}\end{prompt}
\]
