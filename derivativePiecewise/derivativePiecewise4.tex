\begin{exercise}

\outcome{Use limits to find the slope of the tangent line at a point.}
\outcome{Understand the definition of the derivative at a point.}
\outcome{Compute the derivative of a function at a point.}
\outcome{Determine whether a piecewise function is differentiable.}


Let $f$ be defined by

\[
f(x) = \begin{cases} 
	\frac{1}{x-1} \text{ if } x \leq 2\\
	\frac{1}{4}x^3+ax+b  \text{ if } x > 2
\end{cases}
\]

Determine the values of $a$ and $b$ which make $f$ differentiable at $x=2$.

\begin{prompt}
	\begin{align*}
		a&= \answer{-4}\\
		b&= \answer{7}
	\end{align*}
\end{prompt}


\end{exercise}
