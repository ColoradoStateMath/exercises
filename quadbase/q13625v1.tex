\begin{exercise}

Calculate \[\lim_{t\to 4} \frac{2}{t+6}.\]


\begin{hint}
This is a continuous function. Evaluate the function at the limit point
to find the limit value.
\end{hint}


\begin{hint}
\[\lim_{t\to 4} \frac{2}{t+6} = \frac{2}{4+6} = \frac{1}{5}.\]
\end{hint}


\begin{multipleChoice}
\choice{$\infty$}
\choice{DNE}
\choice{$\displaystyle \frac{1}{6}$}
\choice[correct]{$\displaystyle \frac{1}{5}$}
\choice{$\displaystyle \frac{1}{3}$}
\end{multipleChoice}

\end{exercise}
