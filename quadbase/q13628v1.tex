\begin{exercise}

If \[\lim_{x\to 5} f(x) = 10,\] then the limit exists with value 10.


\begin{hint}
This is a tautology.
\end{hint}


\begin{hint}
This is the definition of the notation used.
\end{hint}


\begin{multipleChoice}
\choice{False}
\choice[correct]{True}
\end{multipleChoice}
