\begin{exercise}

Calculate \[\lim_{t\to 0} 5t^2-10t.\]


\begin{hint}
This function is continuous, evaluate the function at the limiting point
to calculate the value of the limit.
\end{hint}


\begin{hint}
\[\lim_{t\to 0} 5t^2-10t = 5(0^2)-10(0) = 0.\]
\end{hint}


\begin{multipleChoice}
\choice{$\infty$}
\choice{-10}
\choice{5}
\choice{DNE}
\choice[correct]{0}
\end{multipleChoice}

\end{exercise}
