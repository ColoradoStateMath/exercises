\begin{exercise}

Calculate \[\lim_{t\to \pi/2} \sin(t).\]


\begin{hint}
This function is continuous. Evaluate it at the point in question,
$t=\pi/2$ to find the limit.
\end{hint}


\begin{hint}
\[\lim_{t\to \pi/2}\sin(t) = \sin(\pi/2) = 1.\]
\end{hint}


\begin{multipleChoice}
\choice{0}
\choice[correct]{1}
\choice{$\infty$}
\choice{DNE}
\choice{-1}
\end{multipleChoice}

\end{exercise}
