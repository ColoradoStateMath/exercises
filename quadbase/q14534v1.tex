\begin{exercise}

Find and identify the extrema of $g(x) = e^{2x}$.


\begin{hint}
Since $g$ is a continuous function on $\mathbb{R}$, find its critical
points and use the second derivative test to identify extrema.
\end{hint}


\begin{hint}
\begin{align*}
g'(x) &= 2e^{2x} \\
g''(x) &= 4e^{2x} \\
g'(x^*)=0 &= 2e^{2x^*} \Rightarrow e^{2x^*} = 0
\end{align*} Since there are no solutions to $e^{2x^*}=0$, there are
no critical points of $g$ and hence no extrema.
\end{hint}


\begin{multipleChoice}
\choice{The function $g$ has one extremum: a minimum at $x= - \infty$.}
\choice{The function $g$ has two extrema: a minimum at $x= -\infty$ and a
maximum at $x=\infty$.}
\choice{The function $g$ has three extrema: a minimum at $x=-\infty$, a minimum
at $x=0$, and a maximum at $x=\infty$.}
\choice[correct]{The function $g$ has no extrema.}
\end{multipleChoice}

\end{exercise}
