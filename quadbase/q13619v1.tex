\begin{exercise}

Calculate \[\lim_{t\to 2} 3t^2-2t-6.\]


\begin{hint}
This function is continuous, evaluate the function at the limiting point
to calculate the value of the limit.
\end{hint}


\begin{hint}
\[\lim_{t \to 2} 3t^2-2t-6 = 3(2^2)-2(2)-6 = 12-4-6 = 2.\]
\end{hint}


\begin{multipleChoice}
\choice{DNE}
\choice[correct]{2}
\choice{$\infty$}
\choice{-4}
\choice{-6}
\end{multipleChoice}

\end{exercise}
