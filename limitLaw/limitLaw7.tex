\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\outcome{Calculate limits by replacing a function with a continuous
  function.}
\tag{limit}

	If you know that $\lim_{x \to 2} f(x) = 4$ and $\lim_{x \to 0} g(x) = 2$, then evaluate the following limit:
	
	\begin{hint}
		 As $x\to 0$, $x+2 \to 2$.
	\end{hint}
	\[
		\lim_{x \to 0} g(x)\sqrt{f(x+2)} = \answer{4}
	\]
	
\end{exercise}
