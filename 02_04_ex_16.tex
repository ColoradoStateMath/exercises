\begin{exercise}

\tag{derivative}
\tag{product rule}

\outcome{Identify products of functions.}
\outcome{Use the product rule to calculate derivatives.}

Find the derivative of $f(x)=x\sin(x)$.
\[
f'(x)
\begin{prompt}
= \answer{\sin(x)+x\cos(x)}
\end{prompt}
\]
