\begin{exercise}
\outcome{Recognize a composition of functions.}
\outcome{Take derivatives of compositions of functions using the chain rule.}
\tag{derivative}
Let $r(z) = -\cot \left(2 \sqrt{2} \sqrt[4]{z}\right)$. Compute:

\[
\dd{z}r(z)
\begin{prompt}=\answer{\frac{\csc ^2\left(2 \sqrt{2} \sqrt[4]{z}\right)}{\sqrt{2} \sqrt[4]{z}^3}}\end{prompt}
\]
\end{exercise}
