\begin{exercise}
\outcome{Recognize a composition of functions.}
\outcome{Take derivatives of compositions of functions using the chain rule.}
\tag{derivative}
Let $y(u) = \cot \left(6 \sqrt{-u}\right)$. Compute:

\[
y'(u)
\begin{prompt}=\answer{\frac{3 \csc ^2\left(6 \sqrt{-u}\right)}{\sqrt{-u}}}\end{prompt}
\]
\end{exercise}
