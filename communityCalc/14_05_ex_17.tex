\tag{directional derivative}

How would you show that the curve $\vec{r}(t)=\vector{\ln(t),t\ln(t),t}$ is tangent to the surface $xz^2-yz+\cos(xy)=1$ at the point $(0,0,1)$?

\begin{multipleChoice}
\choice{Show that $\vec{r}'(1)$ is parallel to the gradient of $xz^2-yz+\cos(xy)$ at $(0,0,1)$.}
\choice{Show that $\vec{r}'(1)$ is perpendicular to the gradient of $xz^2-yz+\cos(xy)$ at $(0,0,1)$.}
\choice{Show that $\vec{r}'(1)$ is equal to the gradient of $xz^2-yz+\cos(xy)$ at $(0,0,1)$.}
\choice[correct]{Show that $\vec{r}'(1)+\vector{0,0,1}$ lies in the plane tangent to $xz^2-yz+\cos(xy)=1$ at the point $(0,0,1)$.}
\end{multipleChoice}
