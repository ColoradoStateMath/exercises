\begin{exercise}

\tag{integral}

Let $R$ be the region between by the curves $y = x^4$ and $y = 24-x^2$

\begin{question}
	Set up, but do not evaluate, an integral representing the area of $R$.
	
	\begin{prompt}
		\textrm{Area} = \answer{\int_{-2}^2 24-x^2-x^4  \d x}
	\end{prompt}
\end{question}

\begin{question}
	Generate a solid by revolving $R$ about the line $y = -1$.  Set up, but do not evaluate, an integral representing the volume of this solid.
	
	\begin{prompt}
		\textrm{Volume} = \answer{\int_{-2}^2 \pi ((25-x^2)^2 - (x^4+1)^2) \d x}
	\end{prompt}
\end{question}

\begin{question}
	Generate a solid by revolving $R$ about the line $x = 5$.  Set up, but do not evaluate, an integral representing the volume of this solid.
	
	\begin{prompt}
		\textrm{Volume} = \answer{\int_{-2}^2 2\pi (5-x)(24-x^2-x^4) \d x}
	\end{prompt}
\end{question}


\begin{question}
	Generate a solid by whose base is $R$ and whose cross sections perpendicular to the $x$ -axis are squares.  Set up, but do not evaluate, an integral representing the volume of this solid.
	
	\begin{prompt}
		\textrm{Volume} = \answer{\int_{-2}^2 (24-x^2-x^4)^2 \d x}
	\end{prompt}
\end{question}


\begin{question}
	Set up, but do not evaluate, an integral which calculates the perimeter of $R$.
	
	\begin{prompt}
		\textrm{Perimeter} = \answer{\int_{-2}^2 \sqrt{1+(4x^3)^2} + \sqrt{1+(-2x)^2} \d x}
	\end{prompt}
\end{question}



\end{exercise}
